Usage

enabled for all axis, you can disabled by set axis.arrow = false, see blow code

yAxis: [{
  lineWidth: 1,
  lineColor: 'black',
  arrow: true,
  reversed: true // reversed axis arrow
}, {
  opposite: true,
  linkedTo: 0,
  lineWidth: 1,
  lineColor: 'red',
  arrow: false // diabled axis arrow
}],

xAxis: [{
  arrow: true,
  lineColor: 'black',
}, {
  // enabled by default
  opposite: true,
  linkedTo: 0
}]
